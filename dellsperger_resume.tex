% Custom commands

% resume item - short title with description
\newcommand{\resumeItemNoDescription}[1]{
  \item\small{
    \textbf{#1\vspace{-5pt}}
  }
}

% resume item - short title with description
\newcommand{\resumeItem}[2]{
  \item\small{
    \textbf{#1}{: #2 \vspace{-5pt}}
  }
}

% resume item - short title with description, then tech stack components
\newcommand{\resumeItemWithTech}[3]{
  \item\small{
    \textbf{#1}{: #2 \vspace{-2pt}}\\
    \textbf{Technology Stack}{: #3 \vspace{-2pt}}
  }
}
